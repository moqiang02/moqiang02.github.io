jquery.validateе’Ңonsubmit()еҗҢж—¶дҪҝз”Ёзҡ„ж–№жі•

жңҖиҝ‘еҶҷдәҶзӮ№еүҚз«Ҝд»Јз ҒпјҢйӘҢиҜҒformиЎЁеҚ•зҡ„ж—¶еҖҷз”ЁеҲ°дәҶjquery validateгҖӮиҝҷдёӘдёңиҘҝеҫҲеҘҪз”ЁпјҢе…·дҪ“зҡ„дҪҝз”Ёж–№жі•е°ұдёҚеҝ…е•°е—ҰдәҶгҖӮ

иҝҷдёӘдёңиҘҝдё»иҰҒжҳҜз”ЁжқҘйӘҢиҜҒиЎЁеҚ•ж•°жҚ®зҡ„ж јејҸпјҢжңүж—¶еҖҷиҷҪз„¶ж јејҸжӯЈзЎ®дәҶдҪҶжҳҜиҝҳиҰҒдҝқиҜҒжҹҗдәӣеӯ—ж®өзҡ„е”ҜдёҖжҖ§пјҢе°ұиҰҒеҸҰеӨ–еҲӨж–ӯдәҶпјҢжҲ‘з”Ёзҡ„ж–№жі•жҳҜеңЁformеҠ дёҖдёӘonsubmitдәӢд»¶гҖӮ

жөӢиҜ•зҡ„ж—¶еҖҷеҸ‘зҺ°пјҢеҸӘиҰҒж»Ўи¶іvalidateйӘҢиҜҒзҡ„ж јејҸпјҢonsubmitзҡ„еҲӨж–ӯдјҡиў«еҝҪз•ҘжҺүгҖӮдәӢе®һдёҠпјҢjquery validateд№ҹжҳҜеңЁsubmitзҡ„ж—¶еҖҷејҖе§ӢеҲӨж–ӯзҡ„пјҢе№¶дё”еңЁonsubmitдәӢд»¶д№ӢеҗҺпјҢжүҖд»Ҙonsubmitиҝ”еӣһfalseпјҢеҸӘиҰҒж»Ўи¶іvalidateзҡ„жқЎд»¶д»Қз„¶дјҡиҝ”еӣһtrueпјҢиЎЁеҚ•д»Қиў«жҸҗдәӨгҖӮдёҚд»…еҰӮжӯӨпјҢжҲ‘еңЁscriptи„ҡжң¬дёӯзҡ„submi()еңЁvalidateйӘҢиҜҒеҗҺдјҡиҮӘеҠЁжү§иЎҢпјҢд№ҹе°ұжҳҜиҜҙпјҢеҚідҪҝйӘҢиҜҒдёҚйҖҡиҝҮпјҢиЎЁеҚ•д»Қз„¶еҸҜд»ҘйҖҡиҝҮsubmit()еҮҪж•°иҮӘеҠЁжҸҗдәӨгҖӮ

иҝҷдёӘж—¶еҖҷе°ұйңҖиҰҒеңЁжһ„йҖ validateи§„еҲҷзҡ„ж—¶еҖҷеҠ дёҠsubmitHandlerгҖӮ

submitHandlerдҪҝз”Ёж–№жі•иҜҰи§Ғпјҡhttp://www.w3cschool.cc/jquery/jquery-plugin-validate.html

jquery validate submitHandler жҸҗдәӨеҜјиҮҙжӯ»еҫӘзҺҜ

domеҜ№еғҸзҡ„жҸҗдәӨform.submit();е’ҢjqueryеҜ№еғҸзҡ„жҸҗдәӨ$('').submit();еҠҹиғҪдёҠжҳҜжІЎжңүд»Җд№ҲеҢәеҲ«зҡ„гҖӮдҪҶжҳҜеҰӮжһңз”ЁдәҶjquery validateжҸ’д»¶пјҢжҸҗдәӨж—¶иҝҷдәҢдёӘе°ұеҢәеҲ«еӨ§дәҶгҖӮ$('').submit();жҸҗдәӨж—¶пјҢjquery validateдјҡиҝӣиЎҢйӘҢиҜҒпјҢsubmitHandlerйҮҢйқўеҰӮжһңеҶҷдәҶ$('').submit();дјҡеҜјиҮҙжӯ»еҫӘзҺҜпјҢиҖҢform.submit();еҲҷдёҚдјҡгҖӮ
иҝҷдёӘй—®йўҳжҳҜдёӘе°Ҹй—®йўҳпјҢеҰӮжһңдёҚжіЁж„ҸпјҢжҲ–иҖ…еҶҷд№ жғҜдәҶпјҢе®№жҳ“еҮәй”ҷпјӣе№¶дё”еҮәй”ҷж—¶пјҢеҫҲйҡҫжҹҘжүҫгҖӮжҲ‘иҠұдәҶиҝ‘дәҢдёӘе°Ҹж—¶пјҢжүҚжүҫеҲ°еҺҹеӣ гҖӮ

    $("#product_form").validate({  
        focusInvalid: false,  
        submitHandler: function(form) {  
            $('.submit').submit();  //jqueryдёӯзҡ„submitеҮҪж•°жңүеҸӮж•°пјҡsubmit(function(){})  
        }  
    });

жӯЈзЎ®еҶҷжі•пјҡ

    $("#product_form").validate({

еҮәзҺ°жӯ»еҫӘзҺҜзҡ„ж—¶еҖҷпјҢжөҸи§ҲеҷЁдјҡдёҖзӣҙеӨ„дәҺеҠ иҪҪдёӯпјҢзӣҙеҲ°жөҸи§ҲеҷЁжҢӮжҺүпјҢиҝҷдёӘж—¶еҖҷпјҢеҸҜд»ҘеҠ дёҖдёӘalertпјҢдҪ е°ұдјҡзҹҘйҒ“пјҢvalidateдёҖзӣҙеңЁеҫӘзҺҜйӘҢиҜҒгҖӮдёӘдәәи§үеҫ—пјҢиҷҪз„¶жҳҜдёӘе°Ҹй—®йўҳпјҢдҪҶжҳҜеҫҲйҡҫжғіиө·жқҘгҖӮ
