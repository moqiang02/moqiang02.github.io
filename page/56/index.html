@Override
public boolean onKeyDown(int keyCode, KeyEvent event) {
    if (keyCode == KeyEvent.KEYCODE_BACK && mWebView.canGoBack()) {
        mWebView.goBack();// 返回前一个页面
        return true;
    }
    return super.onKeyDown(keyCode, event);
