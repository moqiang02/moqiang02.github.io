hexo tagcloud 只显示40个

无意中发现 hexo tagcloud 只显示40个。

找到
/usr/local/lib/node_modules/hexo/lib/plugins/helper/tagcloud.js

修改
amount: 0, // 0 = unlimited

var defaults = {
  min_font: 10,
  max_font: 20,
  unit: 'px',
  //start_color: '',
  //end_color: '',
  //color: false,
  amount: 0, // 0 = unlimited
  orderby: 'name', // name, length, random
  order: 1
  //exclude: []
};
