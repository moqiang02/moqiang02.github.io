        var pro = $("#activity_project").val();  
        $.ajax({  
            type: "POST",  
            url: "${pageContext.request.contextPath}/getMyPlan",  
            data: { //发送给数据库的数据  
                idKey: pro  
            },  
            dataType: 'json',  
            success: function(data) {  
                alert(data);  
                $.each(data,  
                function(index, value) {  
                    alert(data[index].id);  
                    $("#activity_project").append('<option value="data[index].id">data[index].specification</option>');  
                });  
            }  
        })
