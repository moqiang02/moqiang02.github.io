            if (count($params) > 0)

            }

        function init($params = array()) //еҲқе§ӢеҢ–ж•°жҚ®  
        {  
            if (count($params) > 0)  
            {  
                foreach ($params as $key => $val)  
                {  
                    if (isset($this->$key))  
                    {  
                        $this->$key = $val;  
                    }  
                }  
            }

        function create_links()  
        {  
            ///////////////////////////////////////////////////////  
            //еҮҶеӨҮж•°жҚ®  
            ///////////////////////////////////////////////////////  
            $page_url = $this->page_url;  
            $rows_num = $this->rows_num;  
            $page_size = $this->page_size;  
            $links_num = $this->links_num;  

            if ($rows_num == 0 OR $page_size == 0)  
            {  
                return '';  
            }  

            $pages = intval($rows_num/$page_size);  
            if ($rows_num % $page_size)  
            {  
                //жңүдҪҷж•°pages+1  
                $pages++;  
            };  
            $page_num = $this->page_num < 1 ? '1' : $this->page_num;  

            $anchor_class = '';  
            if($this->anchor_class !== '')  
            {  
                $anchor_class = 'class="'.$this->anchor_class.'" ';  
            }  

            $current_class = '';  
            if($this->current_class !== '')  
            {  
                $current_class = 'class="'.$this->current_class.'" ';  
            }  
            if($pages == 1)  
            {  
                return '';  
            }  
            if($links_num < 0)  
            {  
                return '- -пјҒlinks_numеҝ…йЎ»еӨ§дәҺзӯүдәҺ0';  
            }  
            ////////////////////////////////////////////////////////  
            //еҲӣе»әй“ҫжҺҘејҖе§Ӣ  
            ////////////////////////////////////////////////////////  
            $output = $this->full_tag_open;  
            $output .= $this->info_tag_open.'е…ұ'.$rows_num.'жқЎж•°жҚ®  з¬¬ '.$page_num.'/'.$pages.' йЎө'.$this->info_tag_close;  
            //йҰ–йЎө  
            if($page_num > 1)  
            {  
                $output .= $this->first_tag_open.'<a '.$anchor_class.' href="'.site_url($page_url).'" >йҰ–йЎө</a>'.$this->first_tag_close;  
            }  
            //дёҠдёҖйЎө  
            if($page_num > 1)  
            {  
                $n = $page_num - 1;  
                $output .= $this->prev_tag_open.'<a '.$anchor_class.' href="'.site_url($page_url.'/'.$n).'" >дёҠдёҖйЎө</a>'.$this->prev_tag_close;  
            }  
            //pages  
            for($i=1;$i<=$pages;$i++)  
            {  
                $pl = $page_num - $links_num < 0 ? 0 : $page_num - $links_num;  
                $pr = $page_num + $links_num > $pages ? $pages : $page_num + $links_num;  
                //еҲӨж–ӯй“ҫжҺҘдёӘж•°жҳҜеҗҰеӨӘе°‘пјҢдёҫдҫӢпјҢеҒҮи®ҫlinks_num = 2пјҢеҲҷй“ҫжҺҘдёӘж•°дёҚеҸҜе°‘дәҺ 5 дёӘпјҢдё»иҰҒжҳҜ еҪ“page_num зӯүдәҺ 1пјҢ 2 е’Ң nпјҢn-1зҡ„ж—¶еҖҷ  
                if($pr < 2 * $links_num + 1)  
                {  
                    $pr = 2 * $links_num + 1;  
                }  
                if($pl > $pages-2 * $links_num)  
                {  
                    $pl = $pages - 2 * $links_num;  
                }  
                if($i == $page_num)  
                {   //current page  
                    $output .= $this->cur_tag_open.'<span '.$current_class.' >'.$i.'</span>'.$this->cur_tag_close;  
                }else if($i >= $pl && $i <= $pr)  
                {  
                    $output .= $this->num_tag_open.'<a '.$anchor_class.' href="'.site_url($page_url.'/'.$i).'" >'.$i.'</a>'.$this->num_tag_close;  
                }  
            }  
            //дёӢдёҖйЎө  
            if($page_num < $pages)  
            {  
                $n = $page_num + 1;  
                $output .= $this->next_tag_open.'<a '.$anchor_class.' href="'.site_url($page_url.'/'.$n).'" >дёӢдёҖйЎө</a>'.$this->next_tag_close;  
            }  
            //жң«йЎө  
            if($page_num < $pages)  
            {  
                $output .= $this->last_tag_open.'<a '.$anchor_class.' href="'.site_url($page_url.'/'.$pages).'" >жң«йЎө</a>'.$this->last_tag_close;  
            }  

            $output.=$this->full_tag_close;  
            return $output;  
        }  
    }
